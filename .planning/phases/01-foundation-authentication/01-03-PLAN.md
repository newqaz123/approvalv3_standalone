```typescript
import { clerkMiddleware, createRouteMatcher } from '@clerk/nextjs/server'
import { NextResponse } from 'next/server'

const isAdminRoute = createRouteMatcher(['/admin(.*)'])
const isProtectedRoute = createRouteMatcher(['/dashboard(.*)', '/admin(.*)'])

export default clerkMiddleware(async (auth, req) => {
  const { sessionClaims, userId } = await auth()

  // Protect admin routes - only users with 'admin' role can access
  if (isAdminRoute(req)) {
    if (!userId) {
      const signInUrl = new URL('/sign-in', req.url)
      return NextResponse.redirect(signInUrl)
    }
    if (sessionClaims?.metadata?.role !== 'admin') {
      // Non-admin users trying to access admin routes get redirected to dashboard
      return NextResponse.redirect(new URL('/dashboard', req.url))
    }
  }

  // Protect dashboard routes - must be authenticated
  if (isProtectedRoute(req)) {
    const { isAuthenticated, redirectToSignIn } = await auth()
    if (!isAuthenticated) {
      return redirectToSignIn()
    }
  }
})

export const config = {
  matcher: [
    // Skip Next.js internals and all static files
    '/((?!_next|[^?]*\\.(?:html?|css|js(?!on)|jpe?g|webp|png|gif|svg|ttf|woff2?|ico|csv|docx?|xlsx?|zip|webmanifest)).*)',
    // Always run for API routes
    '/(api|trpc)(.*)',
  ],
}
```

Note: This reads role from Clerk's publicMetadata. Role assignment happens when users are created (plan 01-04).

```typescript
import { auth } from '@clerk/nextjs/server'
import { unauthorized } from 'next/navigation'

/**
 * Verifies the current user is authenticated and returns their userId.
 * Throws 401 if not authenticated.
 */
export async function requireUser() {
  const { userId } = await auth()

  if (!userId) {
    unauthorized()
  }

  return userId
}

/**
 * Verifies the current user is an admin.
 * Throws 401 if not authenticated or not an admin.
 * @returns userId of the admin user
 */
export async function requireAdmin() {
  const { userId, sessionClaims } = await auth()

  if (!userId) {
    unauthorized()
  }

  if (sessionClaims?.metadata?.role !== 'admin') {
    unauthorized()
  }

  return userId
}

/**
 * Gets the current user's role from Clerk metadata.
 * Returns null if not authenticated.
 */
export async function getUserRole(): Promise<string | null> {
  const { sessionClaims } = await auth()
  return (sessionClaims?.metadata?.role as string) || null
}

/**
 * Checks if the current user has a specific role.
 * Returns false if not authenticated or role doesn't match.
 */
export async function hasRole(role: string): Promise<boolean> {
  const userRole = await getUserRole()
  return userRole === role
}
```

These functions will be used in Server Actions for user and department management.
