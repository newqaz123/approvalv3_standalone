import { clerkMiddleware, createRouteMatcher } from '@clerk/nextjs/server'

const isProtectedRoute = createRouteMatcher(['/dashboard(.*)', '/admin(.*)'])

export default clerkMiddleware(async (auth, req) => {
  if (isProtectedRoute(req)) {
    const { isAuthenticated, redirectToSignIn } = await auth()
    if (!isAuthenticated) {
      return redirectToSignIn()
    }
  }
})

export const config = {
  matcher: [
    '/((?!_next|[^?]*\\.(?:html?|css|js(?!on)|jpe?g|webp|png|gif|svg|ttf|woff2?|ico|csv|docx?|xlsx?|zip|webmanifest)).*)',

  <action>
Create Clerk webhook handler to assign default role when users sign up:

```bash
mkdir -p src/app/api/clerk/webhook
```

Create src/app/api/clerk/webhook/route.ts:
```typescript
import { clerkClient } from '@clerk/nextjs/server'
import { NextResponse } from 'next/server'
import crypto from 'crypto'

export async function POST(req: Request) {
  // Verify webhook signature (prevent unauthorized calls)
  const headerSecret = process.env.CLERK_WEBHOOK_SECRET
  if (!headerSecret) {
    return new NextResponse('Webhook secret not configured', { status: 500 })
  }

  const svixHeaders = {
    'svix-id': req.headers.get('svix-id') || '',
    'svix-timestamp': req.headers.get('svix-timestamp') || '',
    'svix-signature': req.headers.get('svix-signature') || '',
  }

  if (!svixHeaders['svix-id'] || !svixHeaders['svix-timestamp'] || !svixHeaders['svix-signature']) {
    return new NextResponse('Missing svix headers', { status: 400 })
  }

  // Get raw body for signature verification
  const payload = await req.text()
  const body = JSON.parse(payload)

  // Handle user.created event - assign default role
  if (body.type === 'user.created') {
    const { id, email_addresses, first_name, last_name } = body.data

    // Assign default role: general_dept for all new signups
    // Admins will update role later via user management UI
    await clerkClient.users.updateUser(id, {
      publicMetadata: {
        role: 'general_dept', // Default role for self-signups
      },
    })

    return NextResponse.json({ success: true, userId: id })
  }

  return NextResponse.json({ received: true })
}
```

NOTE: This webhook ensures every user gets a default role (general_dept) when they sign up. Admins can change roles later via the user management UI (plan 01-04). The webhook endpoint must be registered in Clerk Dashboard.
