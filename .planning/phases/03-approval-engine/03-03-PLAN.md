```typescript
/**
 * Update a user's level in the hierarchy
 * Validates no pending approvals, logs change to audit trail
 */
export async function updateUserLevel(
  userId: string,
  newLevel: number,
  expectedUpdatedAt?: Date
): Promise<{ success: boolean; error?: string }> {
  const { userId: adminId } = await auth()
  if (!adminId) {
    throw new Error('Unauthorized')
  }

  // Verify admin role
  const admin = await prisma.user.findUnique({
    where: { id: adminId },
    select: { role: true, name: true },
  })

  if (admin?.role !== 'admin') {
    throw new Error('Admin access required')
  }

  // Get user with department
  const user = await prisma.user.findUnique({
    where: { id: userId },
    select: {
      id: true,
      name: true,
      level: true,
      departmentId: true,
      updatedAt: true,
    },
  })

  if (!user) {
    return { success: false, error: 'User not found' }
  }

  if (!user.departmentId) {
    return { success: false, error: 'User must belong to a department' }
  }

  // Validate no pending approvals in department
  const validation = await validateHierarchyChange(user.departmentId)
  if (!validation.allowed) {
    return { success: false, error: validation.error }
  }

  // Optimistic locking - check updatedAt if provided
  if (expectedUpdatedAt && user.updatedAt > expectedUpdatedAt) {
    return {
      success: false,
      error: 'User was modified by another admin. Please refresh and try again.',
    }
  }

  const oldLevel = user.level || 1

  // Skip if no change
  if (oldLevel === newLevel) {
    return { success: true }
  }

  // Update user level and create audit log in transaction
  await prisma.$transaction([
    // Update user level
    prisma.user.update({
      where: { id: userId },
      data: { level: newLevel },
    }),
    // Log the hierarchy change
    // Using RequestActivity model with special action type for hierarchy changes
    // This keeps all audit logs in one place
    prisma.requestActivity.create({
      data: {
        requestId: 'SYSTEM', // Special ID for non-request activities
        userId: adminId,
        action: 'hierarchy_changed',
        comments: `Changed ${user.name}'s level from ${oldLevel} to ${newLevel} in department`,
      },
    }),
  ])

  revalidatePath(`/admin/departments/${user.departmentId}/hierarchy`)
  revalidatePath('/admin/users')

  return { success: true }
}

/**
 * Get hierarchy change history for a department
 */
export async function getHierarchyChangeHistory(departmentId: string, limit = 10) {
  const { userId } = await auth()
  if (!userId) {
    throw new Error('Unauthorized')
  }

  // Get hierarchy change activities
  const activities = await prisma.requestActivity.findMany({
    where: {
      action: 'hierarchy_changed',
    },
    include: {
      user: {
        select: {
          name: true,
          email: true,
        },
      },
    },
    orderBy: { createdAt: 'desc' },
    take: limit,
  })

  return activities.map(activity => ({
    id: activity.id,
    adminName: activity.user.name,
    change: activity.comments,
    timestamp: activity.createdAt,
  }))
}
```

Note: Using RequestActivity model with action='hierarchy_changed' and requestId='SYSTEM' is a pragmatic approach that keeps all audit logs in one queryable table. A dedicated HierarchyChange model could be added later if needed for compliance.

```typescript
// Update handleDragEnd to call server action directly:
async function handleDragEnd(event: DragEndEvent) {
  const { active, over } = event

  if (!over || isBlocked) return

  const targetLevelStr = over.id.toString()
  if (!targetLevelStr.startsWith('level-')) return

  const targetLevel = parseInt(targetLevelStr.replace('level-', ''), 10)
  const userId = active.id.toString()

  // Find current level
  let currentLevel: number | null = null
  let user: User | null = null
  for (const [level, users] of Object.entries(usersByLevel)) {
    const found = users.find(u => u.id === userId)
    if (found) {
      currentLevel = parseInt(level, 10)
      user = found
      break
    }
  }

  if (!currentLevel || !user || currentLevel === targetLevel) return

  // Optimistic update
  const previousState = { ...usersByLevel }
  setUsersByLevel(prev => {
    const newState = { ...prev }
    newState[currentLevel!] = newState[currentLevel!].filter(u => u.id !== userId)
    newState[targetLevel] = [...(newState[targetLevel] || []), { ...user!, level: targetLevel }]
    return newState
  })

  // Call server action
  try {
    const result = await updateUserLevel(userId, targetLevel)

    if (!result.success) {
      // Rollback on validation error
      setUsersByLevel(previousState)
      // Show error - check if toast is available, otherwise use alert
      if (typeof toast !== 'undefined') {
        toast.error(result.error || 'Failed to update level')
      } else {
        alert(result.error || 'Failed to update level')
      }

      // Re-check if hierarchy is now blocked
      if (result.error?.includes('pending approvals')) {
        setIsBlocked(true)
      }
    } else {
      // Success feedback
      if (typeof toast !== 'undefined') {
        toast.success(`Moved ${user.name} to Level ${targetLevel}`)
      }
    }
  } catch (error) {
    // Rollback on error
    setUsersByLevel(previousState)
    const message = error instanceof Error ? error.message : 'Failed to update level'
    if (typeof toast !== 'undefined') {
      toast.error(message)
    } else {
      alert(message)
    }
  }
}
```
