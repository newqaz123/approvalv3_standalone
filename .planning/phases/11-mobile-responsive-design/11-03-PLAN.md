    - Returns boolean state matching media query
    - Uses useEffect with window.matchMedia
    - Handles addEventListener for 'change' event
    - Cleans up listener on unmount
    - Handles SSR (default to false)

    Implementation:
    ```tsx
    export function useMediaQuery(query: string): boolean {
      const [matches, setMatches] = useState(false)

      useEffect(() => {
        const media = window.matchMedia(query)
        setMatches(media.matches)
        const listener = () => setMatches(media.matches)
        media.addEventListener('change', listener)
        return () => media.removeEventListener('change', listener)
      }, [query])

      return matches
    }
    ```

    Hook is used by RequestDetailModal to choose drawer vs dialog.
  </action>
